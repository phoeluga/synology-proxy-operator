To enable the operator for an Ingress, add this annotation:

  annotations:
    synology.io/enabled: "true"

Example Ingress:

  apiVersion: networking.k8s.io/v1
  kind: Ingress
  metadata:
    name: example
    annotations:
      synology.io/enabled: "true"
  spec:
    rules:
    - host: example.com
      http:
        paths:
        - path: /
          pathType: Prefix
          backend:
            service:
              name: example-service
              port:
                number: 80
